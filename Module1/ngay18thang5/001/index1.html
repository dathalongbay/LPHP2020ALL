== : toán tử so sánh giá trị
        ví dụ '3' và 3 thì vẫn bằng nhau
        === : toán tử so sánh cả kiểu dữ liệu và giá trị
        '3' và 3 thì khác nhau
        != : toán tử so sánh khác giá trị
        !== : toán tử so sánh khác theo cả giá trị và kiểu dữ liệu
        >= : toán tử lớn hơn hoặc bằng
        <= : toán tử nhỏ hơn hoặc bằng
        && : toán tử AND đáp ứng tất cả mệnh đề
        || : toán tử OR chỉ cần đáp ứng 1 mệnh đề
        ! : toán tử phủ định


        ví dụ cú pháp if else đầy đủ
        var age, res;
        if (age > 18) {
            res = 'người trưởng thành';
        } else {
            res = 'trẻ em';
        }

        cú pháp rút gọn
        res = (age > 18) ? 'người trưởng thành' : 'trẻ em';
